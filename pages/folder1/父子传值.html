# 1.props

父组件使用props向子组件传递数

<!-- 父组件 -->
<template>  
  <div>
  <child :msg="message"></child> <!-- 用: 来动态传值 -->
  </div>
</template>

<script>
  import child from './child.vue' //引入子组件
  export default {
    components: { child }, //注册子组件
    data () {
      return {
        message: 'got a message from your father'
      }
    }
  }
</script>
<!-- 子组件 -->
<template> 
<div>{{message}}</div>
</template>

<script>
  export default {
    props: { //使用props接受父组件的值，props: []不限制类型
      msg: {
        type: String,
        required: true
      }
    }
  }
</script>

# 3.eventBus

兄弟组件传值

<!-- 创建Vue实例作为Bus媒介 -->
<script>
  import Vue from 'vue'
  export default eventBus
</script>

在兄弟组件中，都引入Bus文件。分别调用Bus事件触发和监听来实现通信和参数传递。

<!-- 组件1 -->
<template>
  <button @click="toBus">pass</button>
</template>

<script>
  import Bus from '../common/js/bus.js'
  export default {
    methods: {
      toBus () {
        Bus.$emit('on','biu~') //通过$emit触发
      }
    }
  }
</script>
<!-- 组件2 -->
<script>
  import Bus from '../common/js/bus.js'
  export default {
    data() {
      return {
        message: 'me'
      }
    },
    mounted () {
      Bus.$on('on',(msg) => { //使用箭头函数，避免this指向问题
        this.message = msg
      })
    }
  }
</script>
